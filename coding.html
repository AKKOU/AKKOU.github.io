        return odd-even;
    else
        return even-odd;

#include<iostream>

using namespace std;

int main() {
    int people,count=0;
    vector<int> friends;
    cin >> people;
    vector<bool> visted(people,false);

    for(int i=0;i<people;i++){
        int inputf;
        cin >> inputf;
        friends.push_back(inputf);
    }
    
    for(int i=0;i<people;i++){
        if(visted[i]) continue;

        int tmp = i;
        while(!visted[tmp]){
            visted[tmp] = true;
            tmp = friends[tmp];
        }
        count++;
    }

    cout << count;
    return 0;
}

using namespace std;

int main() {
    int people;
    int max=0;
    vector<int> re_a,re_d;
    cin >> people;

    for(int i=0;i<people;i++){
        int a,d;
        cin >> a >> d;
        re_a.push_back(a);
        re_d.push_back(d);
    }

    for(int i=0;i<people;i++){
        for(int j=1;j<people-i;j++){
            int rr1 = re_a[j-1]*re_a[j-1] + re_d[j-1]*re_d[j-1];
            int rr2 = re_a[j]*re_a[j] + re_d[j]*re_d[j];
            if(rr1>rr2){
                int at1 = re_a[j-1];
                int dt1 = re_d[j-1];

                re_a[j-1] = re_a[j];
                re_d[j-1] = re_d[j];
                re_a[j] = at1;
                re_d[j] = dt1;
            }
        }
    }
    cout << re_a[people-2] << " " << re_d[people-2];
    return 0;
}
